package isometric
{
    import flash.geom.Point;
    
    /**
     *    Класс методов для работы с изометрическими велечинами(объектами). 
     */
    public class IsoUtils
    {        
        /**
         *    Конструктор.
         */
        public function IsoUtils()
        {
            
        }
        
        /**
         *    Функция преобразования трехмерных координат в координаты экрана.
         *    pos - точка трехмерных координат.
         *    Возвращает точку экрана.
         */
        public static function isoToScreen(pos:Point3D):Point
        {
            var screenX:Number = pos.x - pos.z;
            var screenY:Number = (pos.x + pos.z) * .524;
            return new Point(screenX, screenY);
        }
        
        /**
         *    Функция преобразования из координат экрана в 3D координаты.
         *    point - точка экранных координат.
         *    Возвращает 3D точку.
         */
        public static function screenToIso(point:Point):Point3D
        {
            var xpos:Number = (point.y / 0.524 - point.x) / 2 + point.x;
            var ypos:Number = 0;
            var zpos:Number = (point.y / 0.524 - point.x) / 2;
            
            return new Point3D(xpos, ypos, zpos);
        }
        
        /**
         *    Функция (центрирования) получения координат объекта, из координатов курсора,
         *    при условии, что курсор находится в зоне обекта.
         *     pos - координаты курсора.
         *  Возвращает координаты объекта.
         */
        public static function centering(pos:Point3D):Point3D
        {
            pos.x = Math.round(pos.x / 48) * 48;
            pos.y = Math.round(pos.y / 48) * 48;
            pos.z = Math.round(pos.z / 48) * 48;
            return pos;
        }
    }
}
