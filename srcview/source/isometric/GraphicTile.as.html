package isometric
{
    import isometric.IsoUtils;
    import isometric.Point3D;
    import flash.geom.Point;
    import spark.components.Image;
    
    /**
     *    Графическая ячейка.
     */
    public class GraphicTile extends IsoObject
    {
        /**
         *    Конструктор.
         */
        public function GraphicTile(size:Number, x:int, y:int, source:String, xOffset:int, yOffset:int):void
        {
            super(size);
        
            //    Преобразуем координаты мыши при броске в изом-ческие.
            var pos:Point3D = IsoUtils.screenToIso(new Point(x, y));
            
            //    Центрируем изображение относительно ячейки.
            pos = IsoUtils.centering(pos);
            
            //    Задаем координаты в мире.
            this.position = pos;
                    
            //    Создаем картинку, задаем ей отступы.
            var img:Image = new Image();
            img.source = source;
            img.x = -xOffset;
            img.y = -yOffset;
            addElement(img);
        }
    }
}
